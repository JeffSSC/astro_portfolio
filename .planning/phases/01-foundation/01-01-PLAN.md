## Wave 2: Theme Toggle

### Task 5: Create Theme Store (SolidJS Island)

**Type:** `auto`
**Files:** `src/stores/theme.ts`

Create theme state management:

```ts
// src/stores/theme.ts
import { createSignal, createEffect } from "solid-js";

type Theme = "light" | "dark";

function getInitialTheme(): Theme {
  if (typeof window === "undefined") return "light";
  
  const stored = localStorage.getItem("theme") as Theme | null;
  if (stored) return stored;
  
  if (window.matchMedia("(prefers-color-scheme: dark)").matches) {
    return "dark";
  }
  return "light";
}

const [theme, setTheme] = createSignal<Theme>(getInitialTheme());

createEffect(() => {
  const current = theme();
  document.documentElement.setAttribute("data-theme", current);
  localStorage.setItem("theme", current);
});

export function toggleTheme() {
  setTheme((prev) => (prev === "light" ? "dark" : "light"));
}

export function getTheme() {
  return theme();
}
```

**Verification:**
- [ ] File created at `src/stores/theme.ts`
- [ ] Exports toggleTheme, getTheme functions
- [ ] localStorage persistence works
- [ ] System preference detection present

### Task 7: Add System Preference Listener

**Type:** `auto`
**Files:** `src/stores/theme.ts`

Add listener for system theme changes:

```ts
// Add to src/stores/theme.ts
if (typeof window !== "undefined") {
  window.matchMedia("(prefers-color-scheme: dark)").addEventListener("change", (e) => {
    const stored = localStorage.getItem("theme");
    if (!stored) {
      setTheme(e.matches ? "dark" : "light");
    }
  });
}
```

**Verification:**
- [ ] Listener added to theme store
- [ ] Responds to OS theme changes
